    <span style="font-size: 14px; color: #ddd; max-width: 800px; text-align: center;">We use cookies to improve your browsing experience, analyze site traffic, and personalize content. By clicking "Accept All Cookies", you agree to the storing of cookies on your device.</span>
    <div style="display: flex; gap: 12px; justify-content: center; flex-wrap: wrap;">
        <button onclick="acceptCookies()" style="background: #00d084; color: #1a1a1a; border: none; padding: 8px 24px; border-radius: 30px; font-weight: 700; cursor: pointer; font-size: 14px; transition: background 0.3s; white-space: nowrap;">Accept All Cookies</button>

<script>
    // Geo-IP Routing Banner Logic
    fetch('https://get.geojs.io/v1/ip/country.json')
        .then(res => res.json())
        .then(data => {
            const country = data.country;
            
            // Only show banner if we haven't already dismissed or accepted
            if (!sessionStorage.getItem('geo_banner_handled')) {
                const banner = document.getElementById('geo-banner');
                const nameSpan = document.getElementById('geo-country-name');
                const progSpan = document.getElementById('geo-programs');
                const linkSpan = document.getElementById('geo-country-link');
                
                let showBanner = false;

                if (country === 'GB') {
                    nameSpan.innerText = 'the UK 🇬🇧';
                    progSpan.innerText = 'UK';
                    linkSpan.href = 'uk/index.html';
                    showBanner = true;
                } else if (country === 'US') {
                    nameSpan.innerText = 'the USA 🇺🇸';
                    progSpan.innerText = 'USA';
                    linkSpan.href = 'us/index.html';
                    showBanner = true;
                } else if (country === 'AE') {
                    nameSpan.innerText = 'the UAE 🇦🇪';
                    progSpan.innerText = 'UAE';
                    linkSpan.href = 'uae/index.html';
                    showBanner = true;
                }

                if (showBanner) {
                    banner.style.display = 'flex';
                    // Optional: Push body down so fixed banner doesn't cover header
                    document.body.style.paddingTop = banner.offsetHeight + 'px';
                    
                    // Handle button clicks to remember state
                    linkSpan.addEventListener('click', function() {
                        sessionStorage.setItem('geo_banner_handled', 'true');
                    });
                }
            }
        })
        .catch(err => console.log('Geo-IP error', err));

    // Cookie Consent Logic
    const cookieConsent = document.getElementById('cookie-consent');
    if (!localStorage.getItem('cookies_accepted')) {
        cookieConsent.style.display = 'flex';
    }

    function acceptCookies() {
        localStorage.setItem('cookies_accepted', 'true');
        cookieConsent.style.display = 'none';
    }

    function declineCookies() {
        localStorage.setItem('cookies_accepted', 'false');
        cookieConsent.style.display = 'none';
    }
</script>
